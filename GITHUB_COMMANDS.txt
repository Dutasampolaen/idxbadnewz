📦 COMPLETE INSTALLATION (Prerequisites + Clone + Install)
────────────────────────────────────────────────────────────────────────────

sudo apt update && \
sudo apt install -y git python3 python3-venv python3-pip curl && \
curl -fsSL https://ollama.com/install.sh | sh && \
ollama pull qwen2.5:7b && \
git clone https://github.com/YOUR_USERNAME/indo-stock-monitor.git && \
cd indo-stock-monitor && \
chmod +x install.sh && \
./install.sh


────────────────────────────────────────────────────────────────────────────
⚙️  POST-INSTALLATION (Configure & Start Services)
────────────────────────────────────────────────────────────────────────────

# 1. Add your API keys
sudo nano /opt/indo_badnews/.env

# 2. Enable and start all services
sudo systemctl enable --now indo_badnews_bot.service indo_badnews.timer indo_volume_screener.timer

# 3. Check status
sudo systemctl status indo_badnews_bot


────────────────────────────────────────────────────────────────────────────
🧪 TESTING COMMANDS
────────────────────────────────────────────────────────────────────────────

# Test news scanner
cd /opt/indo_badnews && source venv/bin/activate && FORCE_RUN=1 python news_watcher.py

# Test volume screener  
cd /opt/indo_badnews && source venv/bin/activate && python volume_screener.py

# View logs
sudo journalctl -u 'indo_*' -f


────────────────────────────────────────────────────────────────────────────
📊 MONITORING COMMANDS
────────────────────────────────────────────────────────────────────────────

# Service status
sudo systemctl status indo_badnews_bot

# Timer schedule
sudo systemctl list-timers indo_*

# Real-time logs
sudo journalctl -u indo_badnews_bot -f

# Recent errors
sudo journalctl -u 'indo_*' -p err --since today


────────────────────────────────────────────────────────────────────────────
🔄 UPDATE COMMANDS (Pull Latest Changes)
────────────────────────────────────────────────────────────────────────────

cd ~/indo-stock-monitor
git pull origin main
sudo cp *.py /opt/indo_badnews/
sudo systemctl restart indo_badnews_bot


────────────────────────────────────────────────────────────────────────────
🗑️  UNINSTALL COMMANDS
────────────────────────────────────────────────────────────────────────────

sudo systemctl stop indo_badnews_bot indo_badnews.timer indo_volume_screener.timer
sudo systemctl disable indo_badnews_bot indo_badnews.timer indo_volume_screener.timer
sudo rm /etc/systemd/system/indo_*
sudo systemctl daemon-reload
sudo rm -rf /opt/indo_badnews
rm -rf ~/indo-stock-monitor

TELEGRAM BOT COMMANDS (After Installation)
================================================================================

/wl                  Show your watchlist
/wl BBRI TLKM       Add tickers to watchlist  
/unwl BBRI          Remove tickers from watchlist
/help               Show help message
